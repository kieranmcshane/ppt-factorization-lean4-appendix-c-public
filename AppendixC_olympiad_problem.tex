\begin{center}
{\Large Theorem C.2 as a self-contained problem}
\end{center}

\begin{problem}
For \(n\ge0\), let \(\NC(n)\) be the set of noncrossing partitions of
\(\{1,\ldots,n\}\), i.e. partitions with no \(a<b<c<d\) such that
\(a,c\) lie in one block and \(b,d\) lie in another.  For a block \(B\), put
\[
  \delta(B)=2\left\lfloor\frac{|B|-1}{2}\right\rfloor,
  \qquad
  \delta(\pi)=\sum_{B\in\pi}\delta(B).
\]
For indeterminates \(\lambda,t\), define
\[
  M_n(\lambda,t)
  =
  \sum_{\pi\in\NC(n)}
    \lambda^{|\pi|}t^{\delta(\pi)/2}
  \in \mathbb Q[\lambda,t].
\]
For \(m\ge0\), set
\[
  H_m(\lambda,t)
  =
  \bigl(M_{i+j+1}(\lambda,t)\bigr)_{0\le i,j\le m}.
\]
Define \(f_{0,m},f_{1,m}\in\mathbb Q[\lambda]\) by
\[
  f_{0,m}(\lambda)=[t^0]\det H_m(\lambda,t),
  \qquad
  f_{1,m}(\lambda)=[t^1]\det H_m(\lambda,t).
\]
Finally define \(d_r(\lambda)\) by
\[
  d_0(\lambda)=1,\qquad
  d_1(\lambda)=\lambda,\qquad
  d_{r+2}(\lambda)=\lambda d_{r+1}(\lambda)-\lambda d_r(\lambda),
\]
and define \(g_m(\lambda)\in\mathbb Q[\lambda]\) by
\[
  d_{m+1}(\lambda)
  =
  \lambda^{\lceil (m+1)/2\rceil}g_m(\lambda),
  \qquad g_m(0)\ne0.
\]

\begin{theoremC}
For every \(m\ge0\),
\[
  g_m(\lambda)\mid
  \bigl(f_{1,m}(\lambda)-f_{0,m}'(\lambda)\bigr)
  \qquad\text{in }\mathbb Q[\lambda].
\]
\end{theoremC}

Here \(t\) is the bookkeeping variable for the first finite-size defect in
Appendix C.  Thus \(M_n(\lambda,0)\) is the balanced moment, \(f_{0,m}\) is the
balanced Hankel determinant, \(f_{1,m}\) is its first finite-size correction,
and \(g_m\) is the threshold factor obtained from the tridiagonal determinant
\(d_{m+1}\).  With these identifications, the displayed divisibility is exactly
Theorem C.2.
\end{problem}
