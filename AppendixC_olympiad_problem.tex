\begin{center}
{\Large A self-contained formulation of Theorem C.2}
\end{center}

\begin{problem}
For \(n\ge0\), define
\[
  M_n(\lambda)
  =
  \sum_{r=0}^{\lfloor n/2\rfloor}
    \binom{n}{2r}\Cat(r)\lambda^{\,n-r},
  \qquad
  \Cat(r)=\frac{1}{r+1}\binom{2r}{r}.
\]
Define \(M_n^{(1)}(\lambda)\) by
\[
\begin{aligned}
M_n^{(1)}(\lambda)
&=
\lambda
\sum_{\substack{a,b,c\ge0\\a+b+c+3=n}}
  (a+1)M_a(\lambda)M_b(\lambda)M_c(\lambda) \\
&\quad+
\lambda
\sum_{\substack{a,b,c,d\ge0\\a+b+c+d+4=n}}
  (a+1)M_a(\lambda)M_b(\lambda)M_c(\lambda)M_d(\lambda).
\end{aligned}
\]
For \(m\ge0\), set
\[
  H_m(\lambda)=\bigl(M_{i+j+1}(\lambda)\bigr)_{0\le i,j\le m},
  \qquad
  H_m^{(1)}(\lambda)
  =
  \bigl(M_{i+j+1}^{(1)}(\lambda)\bigr)_{0\le i,j\le m},
\]
and
\[
  f_{0,m}(\lambda)=\det H_m(\lambda),
  \qquad
  f_{1,m}(\lambda)
  =
  \Tr\!\bigl(\adj(H_m(\lambda))H_m^{(1)}(\lambda)\bigr).
\]
Finally define \(d_r(\lambda)\) by
\[
  d_0(\lambda)=1,\qquad
  d_1(\lambda)=\lambda,\qquad
  d_{r+2}(\lambda)=\lambda d_{r+1}(\lambda)-\lambda d_r(\lambda),
\]
and define \(g_m(\lambda)\) by
\[
  d_{m+1}(\lambda)=
  \lambda^{\lceil (m+1)/2\rceil}g_m(\lambda),
  \qquad g_m(0)\ne0.
\]

\begin{theoremC}
For every \(m\ge0\),
\[
  g_m(\lambda)\mid
  \bigl(f_{1,m}(\lambda)-f_{0,m}'(\lambda)\bigr)
  \qquad\text{in }\mathbb Q[\lambda].
\]
\end{theoremC}
\end{problem}
