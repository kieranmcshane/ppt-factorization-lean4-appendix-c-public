{\Large Theorem C.2 as a problem}
\end{center}

\begin{problem}
Use the notation of Appendix C.  Thus \(f_{0,m}\) denotes the balanced
Hankel determinant, \(f_{1,m}\) denotes the first finite-size correction
polynomial, and \(g_m\) denotes the non-zero threshold factor defined by
\[
  d_{m+1}(\lambda)
  =
  \lambda^{\lceil (m+1)/2\rceil}g_m(\lambda),
  \qquad g_m(0)\ne0.
\]

\begin{theoremC}
For every \(m\ge0\),
\[
  g_m(\lambda)\mid
  \bigl(f_{1,m}(\lambda)-f_{0,m}'(\lambda)\bigr)
  \qquad\text{in }\mathbb Q[\lambda].
\]
\end{theoremC}
\end{problem}
